          if (isNaN(parseFloat(answer1))){
          do {answer1 = prompt("You did not enter a number. Please enter a first number.");}
          while(isNaN(parseFloat(answer1)));}

          var answer2 = prompt("Please enter a second number");
          if(isNaN(parseFloat(answer2))){
          do answer2 = prompt("You did not enter a number. Please enter a second number.");
          while(isNaN(parseFloat(answer2)));}

          if(answer1==answer2){
          alert("The numbers are equal.");}
          if(answer1>answer2){
          alert(answer1 + " is greater than " + answer2);}
          else{alert(answer1 + " is less than or equal to " + answer2);}
        }
